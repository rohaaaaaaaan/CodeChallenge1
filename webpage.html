Running Time Function

def running_time(start_hour, start_minutes):
    easy_pace = 8 * 60 + 15
    tempo_pace = 7 * 60 + 12
    total_time_seconds = easy_pace * 2 + (tempo_pace * 3)
    total_minutes = total_time_seconds // 60
    total_seconds = total_time_seconds % 60
    end_minutes = total_minutes + start_minutes
    end_hour = start_hour
    if end_minutes >= 60:
        end_hour += end_minutes // 60
        end_minutes = end_minutes % 60
    return end_hour, end_minutes

end_hour, end_minutes = running_time(6, 52)
print(f"You will arrive home for breakfast at {end_hour}:{end_minutes} AM.")

Right Justify Function

def print_twice(message):
    print(message)
    print(message)

    f(v)
    f(v)

do_twice(print_twice, 'spam')

def do_four(f, v):
    do_twice(f, v)
    do_twice(f, v)

do_four(print_twice, 'spam')

Is Triangle Function

def is_triangle(a, b, c):
    if a + b > c and a + c > b and b + c > a:
        print("Yes")
    else:
        print("No")

Turtle: Draw Single Square

import turtle

def square(t, length):
    for _ in range(4):
        t.forward(length)
        t.right(90)

screen = turtle.Screen()
screen.bgcolor("white")

bob = turtle.Turtle()
bob.shape("turtle")
bob.color("blue")

square(bob, 100)

bob.hideturtle()
screen.exitonclick()

Turtle: Circles

import turtle

def polygon(t, length, n):
    angle = 360 / n
    for _ in range(n):
        t.forward(length)
        t.left(angle)

def circle(t, r):
    n = 360
    length = 2 * 3.14159 * r / n
    polygon(t, length, n)

bob = turtle.Turtle()
radii = [50, 100, 150, 200]

for r in radii:
    bob.penup()
    bob.goto(0, -r)
    bob.pendown()
    circle(bob, r)
    bob.penup()

turtle.done()

Turtle: Arcs

import turtle

def polygon(t, length, n):
    angle = 360 / n
    for _ in range(n):
        t.forward(length)
        t.left(angle)

def arc(t, r, angle):
    n = 360
    length = 2 * 3.14159 * r / n
    sides_to_draw = int(n * (angle / 360))
    for _ in range(sides_to_draw):
        t.forward(length)
        t.left(360 / n)

bob = turtle.Turtle()
radii_angles = [(50, 90), (100, 180), (150, 45), (200, 270)]

for r, angle in radii_angles:
    bob.penup()
    bob.goto(0, -r)
    bob.pendown()
    arc(bob, r, angle)
    bob.penup()

turtle.done()
